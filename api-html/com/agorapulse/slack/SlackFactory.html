059 /**
060  * Creates all the necessary beans in the Micronaut context.
061  */
062 @Factory
063 public class SlackFactory {
064 
065     @Bean
066     @Singleton
067     public SlackConfig slackConfig() {
068         return SlackConfig.DEFAULT;
069     }
070 
071     @Bean
072     @Singleton
073     public SlackHttpClient slackHttpClient(SlackConfig config) {
074         return SlackHttpClient.buildSlackHttpClient(config);
075     }
076 
077     @Bean
078     @Singleton
079     public Slack slack() {
080         return Slack.getInstance();
081     }
082 
083     @Bean
084     @Singleton
085     public MethodsClient methodsClient(Slack slack, SlackConfiguration configuration) {
086         if (StringUtils.isNotEmpty(configuration.getSingleTeamBotToken())) {
087             return slack.methods(configuration.getSingleTeamBotToken());
088         }
089         return slack.methods();
090     }
091 
092     @Bean
093     @Singleton
094     public AsyncMethodsClient asyncMethodsClientx(Slack slack, SlackConfiguration configuration) {
095         if (StringUtils.isNotEmpty(configuration.getSingleTeamBotToken())) {
096             return slack.methodsAsync(configuration.getSingleTeamBotToken());
097         }
098         return slack.methodsAsync();
099     }
100 
101     @Bean
102     @Singleton
103     public InstallationService installationService(SlackConfiguration slackConfiguration) {
104         return new FileInstallationService(slackConfiguration);
105     }
106 
107     @Bean
108     @Singleton
109     public InstallationEnumerationService installationEnumerationService(InstallationService installationService, SlackConfiguration slackConfiguration) {
110         if (installationService instanceof FileInstallationService || installationService instanceof ObservableInstallationService && ((ObservableInstallationService) installationService).getDelegateType().equals(FileInstallationService.class)) {
111             return new FileInstallationEnumerationService(slackConfiguration, FileInstallationEnumerationService.DEFAULT_ROOT_DIR, false);
112         }
113 
114         return null;
115     }
116 
117     @Bean
118     @Singleton
119     public DistributedAppMethodsClientFactory distributedAppMethodsClientFactory(Slack slack, InstallationService service) {
120         return new DistributedAppMethodsClientFactory() {
121             @Override
122             public Optional<MethodsClient> createClient(String enterpriseId, String methodsId) {
123                 Bot bot = service.findBot(enterpriseId, methodsId);
124                 if (bot != null) {
125                     return Optional.of(slack.methods(bot.getBotAccessToken(), bot.getTeamId()));
126                 }
127 
128                 return Optional.empty();
129             }
130 
131             @Override
132             public Optional<MethodsClient> createClient(Bot bot) {
133                 return Optional.of(slack.methods(bot.getBotAccessToken(), bot.getTeamId()));
134             }
135         };
136     }
137 
138     @Bean
139     @Singleton
140     public DistributedAppAsyncMethodsClientFactory distributedAppAsyncMethodsClientFactory(Slack slack, InstallationService service) {
141         return new DistributedAppAsyncMethodsClientFactory() {
142             @Override
143             public Optional<AsyncMethodsClient> createClient(String enterpriseId, String methodsId) {
144                 Bot bot = service.findBot(enterpriseId, methodsId);
145                 if (bot != null) {
146                     return Optional.of(slack.methodsAsync(bot.getBotAccessToken(), bot.getTeamId()));
147                 }
148 
149                 return Optional.empty();
150             }
151 
152             @Override
153             public Optional<AsyncMethodsClient> createClient(Bot bot) {
154                 return Optional.of(slack.methodsAsync(bot.getBotAccessToken(), bot.getTeamId()));
155             }
156 
157         };
158     }

160     @Bean
161     @Context
162     public App app(
163         SlackConfiguration configuration,
164         Slack slack,
165         InstallationService installationService,
166         List<MicronautAttachmentActionHandler> attachmentActionHandlers,
167         List<MicronautBlockActionHandler> blockActionHandlers,
168         List<MicronautBlockSuggestionHandler> blockSuggestionHandlers,
169         List<MicronautDialogCancellationHandler> dialogCancellationHandlers,
170         List<MicronautDialogSubmissionHandler> dialogSubmissionHandlers,
171         List<MicronautDialogSuggestionHandler> dialogSuggestionHandlers,
172         List<MicronautGlobalShortcutHandler> globalShortcutHandlers,
173         List<MicronautMessageShortcutHandler> messageShortcutHandlers,
174         List<MicronautSlashCommandHandler> slashCommandHandlers,
175         List<MicronautViewClosedHandler> viewClosedHandlers,
176         List<MicronautViewSubmissionHandler> viewSubmissionHandlers,
177         List<MicronautWorkflowStepEditHandler> workflowStepEditHandlers,
178         List<MicronautWorkflowStepExecuteHandler> workflowStepExecuteHandlers,
179         List<MicronautWorkflowStepSaveHandler> workflowStepSaveHandlers,
180         List<MicronautBoltEventHandler<Event>> boltEventHandlers
181     ) {
182         App app = createApp(configuration, slack);
183 
184         attachmentActionHandlers.forEach(h -> app.attachmentAction(h.getCallbackIdPattern(), h));
185         blockActionHandlers.forEach(h -> app.blockAction(h.getActionIdPattern(), h));
186         blockSuggestionHandlers.forEach(h -> app.blockSuggestion(h.getActionIdPattern(), h));
187         dialogCancellationHandlers.forEach(h -> app.dialogCancellation(h.getCallbackIdPattern(), h));
188         dialogSubmissionHandlers.forEach(h -> app.dialogSubmission(h.getCallbackIdPattern(), h));
189         dialogSuggestionHandlers.forEach(h -> app.dialogSuggestion(h.getCallbackIdPattern(), h));
190         globalShortcutHandlers.forEach(h -> app.globalShortcut(h.getCallbackIdPattern(), h));
191         messageShortcutHandlers.forEach(h -> app.messageShortcut(h.getCallbackIdPattern(), h));
192         slashCommandHandlers.forEach(h -> app.command(h.getCommandIdPattern(), h));
193         viewClosedHandlers.forEach(h -> app.viewClosed(h.getCallbackIdPattern(), h));
194         viewSubmissionHandlers.forEach(h -> app.viewSubmission(h.getCallbackIdPattern(), h));
195         workflowStepEditHandlers.forEach(h -> app.workflowStepEdit(h.getCallbackIdPattern(), h));
196         workflowStepExecuteHandlers.forEach(h -> app.workflowStepExecute(h.getPattern(), h));
197         workflowStepSaveHandlers.forEach(h -> app.workflowStepSave(h.getCallbackIdPattern(), h));
198         boltEventHandlers.forEach(h -> app.event(h.getEventType(), h));
199 
200         app.service(installationService);
201 
202         return app;
203     }
204 
205     private static App createApp(SlackConfiguration configuration, Slack slack) {
206         configuration.setSlack(slack);
207 
208         if (configuration.isOAuthInstallPathEnabled()) {
209             return new App(configuration).asOAuthApp(true);
210         }
211 
212         if (configuration.isOpenIDConnectEnabled()) {
213             return new App(configuration).asOpenIDConnectApp(true);
214         }
215 
216         return new App(configuration);
217     }
