235     @Bean
236     @Context
237     public App app(
238         SlackConfiguration configuration,
239         Slack slack,
240         InstallationService installationService,
241         OAuthStateService oAuthStateService,
242         List<MicronautAttachmentActionHandler> attachmentActionHandlers,
243         List<MicronautBlockActionHandler> blockActionHandlers,
244         List<MicronautBlockSuggestionHandler> blockSuggestionHandlers,
245         List<MicronautDialogCancellationHandler> dialogCancellationHandlers,
246         List<MicronautDialogSubmissionHandler> dialogSubmissionHandlers,
247         List<MicronautDialogSuggestionHandler> dialogSuggestionHandlers,
248         List<MicronautGlobalShortcutHandler> globalShortcutHandlers,
249         List<MicronautMessageShortcutHandler> messageShortcutHandlers,
250         List<MicronautSlashCommandHandler> slashCommandHandlers,
251         List<MicronautViewClosedHandler> viewClosedHandlers,
252         List<MicronautViewSubmissionHandler> viewSubmissionHandlers,
253         List<MicronautWorkflowStepEditHandler> workflowStepEditHandlers,
254         List<MicronautWorkflowStepExecuteHandler> workflowStepExecuteHandlers,
255         List<MicronautWorkflowStepSaveHandler> workflowStepSaveHandlers,
256         List<MicronautBoltEventHandler<Event>> boltEventHandlers
257     ) {
258         App app = createApp(configuration, slack);
259 
260         attachmentActionHandlers.forEach(h -> app.attachmentAction(h.getCallbackIdPattern(), h));
261         blockActionHandlers.forEach(h -> app.blockAction(h.getActionIdPattern(), h));
262         blockSuggestionHandlers.forEach(h -> app.blockSuggestion(h.getActionIdPattern(), h));
263         dialogCancellationHandlers.forEach(h -> app.dialogCancellation(h.getCallbackIdPattern(), h));
264         dialogSubmissionHandlers.forEach(h -> app.dialogSubmission(h.getCallbackIdPattern(), h));
265         dialogSuggestionHandlers.forEach(h -> app.dialogSuggestion(h.getCallbackIdPattern(), h));
266         globalShortcutHandlers.forEach(h -> app.globalShortcut(h.getCallbackIdPattern(), h));
267         messageShortcutHandlers.forEach(h -> app.messageShortcut(h.getCallbackIdPattern(), h));
268         slashCommandHandlers.forEach(h -> app.command(h.getCommandIdPattern(), h));
269         viewClosedHandlers.forEach(h -> app.viewClosed(h.getCallbackIdPattern(), h));
270         viewSubmissionHandlers.forEach(h -> app.viewSubmission(h.getCallbackIdPattern(), h));
271         workflowStepEditHandlers.forEach(h -> app.workflowStepEdit(h.getCallbackIdPattern(), h));
272         workflowStepExecuteHandlers.forEach(h -> app.workflowStepExecute(h.getPattern(), h));
273         workflowStepSaveHandlers.forEach(h -> app.workflowStepSave(h.getCallbackIdPattern(), h));
274         boltEventHandlers.forEach(h -> app.event(h.getEventType(), h));
275 
276         app.service(installationService);
277         app.service(oAuthStateService);
278 
279         return app;
280     }

282     private static App createApp(SlackConfiguration configuration, Slack slack) {
283         configuration.setSlack(slack);
284 
285         if (configuration.isOAuthInstallPathEnabled()) {
286             return new App(configuration).asOAuthApp(true);
287         }
288 
289         if (configuration.isOpenIDConnectEnabled()) {
290             return new App(configuration).asOpenIDConnectApp(true);
291         }
292 
293         return new App(configuration);
294     }

296     private <T> T createServiceIfS3Configured(SlackConfiguration configuration, AmazonS3 slackS3, AmazonS3 defaultS3, Function<SlackConfiguration, T> defaultService, BiFunction<String, AmazonS3, T> creator) {
297         if (StringUtils.isEmpty(configuration.getBucket())) {
298             return defaultService.apply(configuration);
299         }
300 
301         AmazonS3 s3 = slackS3 == null ? defaultS3 : slackS3;
302 
303         if (s3 == null) {
304             return  defaultService.apply(configuration);
305         }
306 
307         return creator.apply(configuration.getBucket(), s3);
308     }
