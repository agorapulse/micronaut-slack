18  package com.agorapulse.slack.http;
19  
20  import com.slack.api.bolt.App;
21  import com.slack.api.bolt.request.Request;
22  import io.micronaut.http.HttpRequest;
23  import io.micronaut.http.HttpResponse;
24  import io.micronaut.http.MediaType;
25  import io.micronaut.http.annotation.Body;
26  import io.micronaut.http.annotation.Controller;
27  import io.micronaut.http.annotation.Get;
28  import io.micronaut.http.annotation.Post;
29  
30  /**
31   * The default Web controller that works in Micronaut apps.
32   * This component requires singleton {@link App} instance managed by the Micronaut DI container.
33   *
34   * @see <a href="https://guides.micronaut.io/creating-your-first-micronaut-app/guide/index.html">The official tutorial</a>
35   * @see <a href="https://docs.micronaut.io/latest/api/io/micronaut/http/annotation/Controller.html">@Controller annotation</a>
36   *
37   * This is a copy of <code>com.slack.api.bolt.micronaut.SlackAppController</code> that guarantees compatibility with
38   * older versions of Micronaut.
39   */
40  @Controller("/slack")
41  public class SlackAppController {
42  
43      private final App slackApp;
44      private final SlackAppMicronautAdapter adapter;
45  
46      public SlackAppController(App slackApp, SlackAppMicronautAdapter adapter) {
47          this.slackApp = slackApp;
48          this.adapter = adapter;
49      }
50  
51      @Post(value = "/events", consumes = {MediaType.APPLICATION_FORM_URLENCODED, MediaType.APPLICATION_JSON})
52      public HttpResponse<String> events(HttpRequest<String> request, @Body String body) throws Exception {
53          return adapt(request, body);
54      }
55  
56      @Get("/install")
57      public HttpResponse<String> install(HttpRequest<String> request) throws Exception {
58          if (!slackApp.config().isOAuthInstallPathEnabled()) {
59              return HttpResponse.notFound();
60          }
61          return adapt(request, null);
62      }
63  
64      @Get("/oauth_redirect")
65      public HttpResponse<String> oauthRedirect(HttpRequest<String> request) throws Exception {
66          if (!slackApp.config().isOAuthRedirectUriPathEnabled()) {
67              return HttpResponse.notFound();
68          }
69          return adapt(request, null);
70      }
71  
72      private HttpResponse<String> adapt(HttpRequest<String> request, String body) throws Exception {
73          Request<?> slackRequest = adapter.toSlackRequest(request, body);
74          return adapter.toMicronautResponse(slackApp.run(slackRequest));
75      }
76  
77  }
